using namespace std;
void FastIO() { ios_base::sync_with_stdio(false); cin.tie(NULL); cout.tie(NULL); }

const int MAX = 1000 + 4;
const int DIR[4][2]
{
    {1, 0},
    {-1, 0},
    {0, 1},
    {0, -1}
};
int row, col;
int board[MAX][MAX];
bool isReachEnd;
void DFS(const int cy, const int cx)
{
    board[cy][cx] = 0;
    if (cy == row)
    {
        isReachEnd = true;
        return;
    }
    for (int i = 0; i < 4; ++i)
    {
        const int ny = cy + DIR[i][0];
        const int nx = cx + DIR[i][1];
        if (board[ny][nx] == 1)
            DFS(ny, nx);
    }
}
int main()
{
    FastIO();
    cin >> row >> col;
    string s;
    s.reserve(col);
    for (int i = 1; i <= row; ++i)
    {
        cin >> s;
        for (int j = 1; j <= col; ++j)
        {
            if (s[j - 1] == '0')
                board[i][j] = 1;
            else
                board[i][j] = 0;
        }
    }

    for (int j = 1; j <= col; ++j)
    {
        if (board[1][j] == 1)
            DFS(1, j);
    }

    if (isReachEnd)
        cout << "YES";
    else
        cout << "NO";

    return 0;
}
